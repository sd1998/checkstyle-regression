40      @Test
41      public void testDefault() throws Exception {
42          final DefaultConfiguration checkConfig =
43                  createModuleConfig(StaticMethodCandidateCheck.class);
44          final String[] expected = {
45              "37:5: " + getCheckMessage(MSG_KEY, "foo11"),
46              "56:9: " + getCheckMessage(MSG_KEY, "nestedFoo1"),
47              "86:9: " + getCheckMessage(MSG_KEY, "doSomething"),
48              "123:5: " + getCheckMessage(MSG_KEY, "main"),
49              "176:5: " + getCheckMessage(MSG_KEY, "bar"),
50              "184:5: " + getCheckMessage(MSG_KEY, "fooBar"),
51              "186:5: " + getCheckMessage(MSG_KEY, "barFoo"),
52              "188:5: " + getCheckMessage(MSG_KEY, "fooo"),
53              "190:5: " + getCheckMessage(MSG_KEY, "baar"),
54              "192:5: " + getCheckMessage(MSG_KEY, "fOo"),
55              "194:5: " + getCheckMessage(MSG_KEY, "foO"),
56          };
57          verify(checkConfig, getPath("InputStaticMethodCandidateCheck.java"), expected);
58      }
