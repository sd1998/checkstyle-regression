24  import org.testng.annotations.DataProvider;
25  import org.testng.annotations.Test;
26  
27  import static org.testng.Assert.assertEquals;
28  
29  /*
30   * @test
31   * @bug 8077559
32   * @summary Tests Compact String. This one is for String.subString.
33   * @run testng/othervm -XX:+CompactStrings SubString
34   * @run testng/othervm -XX:-CompactStrings SubString
35   */
36  
37  public class SubString extends CompactString {
38  
39      @DataProvider
40      public Object[][] provider() {
41          return new Object[][] {
42  
43                  new Object[] { STRING_EMPTY, 0, 0, "" },
44                  new Object[] { STRING_L1, 0, 1, "A" },
45                  new Object[] { STRING_L1, 1, 1, "" },
46                  new Object[] { STRING_L2, 0, 2, "AB" },
47                  new Object[] { STRING_L2, 1, 2, "B" },
48                  new Object[] { STRING_LLONG, 0, 8, "ABCDEFGH" },
49                  new Object[] { STRING_LLONG, 7, 8, "H" },
50                  new Object[] { STRING_LLONG, 8, 8, "" },
51                  new Object[] { STRING_LLONG, 3, 7, "DEFG" },
52                  new Object[] { STRING_U1, 0, 1, "\uFF21" },
53                  new Object[] { STRING_U1, 1, 1, "" },
54                  new Object[] { STRING_U1, 0, 0, "" },
55                  new Object[] { STRING_U2, 0, 2, "\uFF21\uFF22" },
56                  new Object[] { STRING_U2, 1, 2, "\uFF22" },
57                  new Object[] { STRING_U2, 2, 2, "" },
58                  new Object[] { STRING_U2, 0, 2, "\uFF21\uFF22" },
59                  new Object[] { STRING_U2, 1, 2, "\uFF22" },
60                  new Object[] { STRING_M12, 1, 2, "A" },
61                  new Object[] { STRING_M11, 0, 1, "A" },
62                  new Object[] { STRING_M11, 1, 2, "\uFF21" },
63                  new Object[] { STRING_UDUPLICATE, 1, 5,
64                          "\uFF22\uFF21\uFF22\uFF21" },
65                  new Object[] { STRING_MDUPLICATE1, 9, 10, "A" },
66                  new Object[] { STRING_MDUPLICATE1, 7, 8, "A" }, };
67      }
68  
69      @Test(dataProvider = "provider")
70      public void testSubstring(String str, int beginIndex, int endIndex,
71              String expected) {
72          map.get(str)
73                  .forEach(
74                          (source, data) -> {
75                              assertEquals(
76                                      data.substring(beginIndex, endIndex),
77                                      expected,
78                                      String.format(
79                                              "testing String(%s).substring(%d, %d), source : %s, ",
80                                              escapeNonASCIIs(data), beginIndex,
81                                              endIndex, source));
82                          });
83      }
84  }
