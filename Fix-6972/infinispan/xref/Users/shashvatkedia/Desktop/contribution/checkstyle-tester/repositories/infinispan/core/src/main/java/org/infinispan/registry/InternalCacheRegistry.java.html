1   package org.infinispan.registry;
2   
3   import java.util.EnumSet;
4   import java.util.Set;
5   
6   import org.infinispan.configuration.cache.Configuration;
7   import org.infinispan.factories.scopes.Scope;
8   import org.infinispan.factories.scopes.Scopes;
9   
10  /**
11   * InternalCacheRegistry. Components which create caches for internal use should use this class to
12   * create/retrieve them
13   *
14   * @author Tristan Tarrant
15   * @since 7.2
16   */
17  @Scope(Scopes.GLOBAL)
18  public interface InternalCacheRegistry {
19     enum Flag {
20        EXCLUSIVE, // means that the cache must be declared only once
21        USER, // means that this cache is visible to users
22     }
23  
24     void registerInternalCache(String name, Configuration configuration);
25  
26     void registerInternalCache(String name, Configuration configuration, EnumSet<Flag> flags);
27  
28     boolean isInternalCache(String name);
29  
30     Set<String> getInternalCacheNames();
31  
32     void filterInternalCaches(Set<String> names);
33  }
