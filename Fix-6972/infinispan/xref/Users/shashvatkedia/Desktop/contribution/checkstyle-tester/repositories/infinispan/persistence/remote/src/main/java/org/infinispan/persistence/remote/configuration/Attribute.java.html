1   package org.infinispan.persistence.remote.configuration;
2   
3   import java.util.HashMap;
4   import java.util.Map;
5   
6   /**
7    * Enumerates the attributes used by the Remote cache store configuration
8    *
9    * @author Tristan Tarrant
10   * @since 5.2
11   */
12  public enum Attribute {
13     // must be first
14     UNKNOWN(null),
15  
16     ASYNC_EXECUTOR("async-executor"),
17     BALANCING_STRATEGY("balancing-strategy"),
18     CONNECT_TIMEOUT("connect-timeout"),
19     EXHAUSTED_ACTION("exhausted-action"),
20     FACTORY("factory"),
21     FORCE_RETURN_VALUES("force-return-values"),
22     HOST("host"),
23     HOTROD_WRAPPING("hotrod-wrapping"),
24     MARSHALLER("marshaller"),
25     MAX_ACTIVE("max-active"),
26     MAX_IDLE("max-idle"),
27     MAX_TOTAL("max-total"),
28     MIN_EVICTABLE_IDLE_TIME("min-evictable-idle-time"),
29     MIN_IDLE("min-idle-time"),
30     KEY_SIZE_ESTIMATE("key-size-estimate"),
31     OUTBOUND_SOCKET_BINDING("outbound-socket-binding"),
32     PING_ON_STARTUP("ping-on-start"),
33     PORT("port"),
34     PROTOCOL_VERSION("protocol-version"),
35     RAW_VALUES("raw-values"),
36     REMOTE_CACHE_NAME("cache"),
37     SOCKET_TIMEOUT("socket-timeout"),
38     TCP_NO_DELAY("tcp-no-delay"),
39     TEST_WHILE_IDLE("test-idle"),
40     TIME_BETWEEN_EVICTION_RUNS("eviction-interval"),
41     TRANSPORT_FACTORY("transport-factory"),
42     VALUE_SIZE_ESTIMATE("value-size-estimate"),
43     ;
44  
45     private final String name;
46  
47     private Attribute(final String name) {
48        this.name = name;
49     }
50  
51     /**
52      * Get the local name of this element.
53      *
54      * @return the local name
55      */
56     public String getLocalName() {
57        return name;
58     }
59  
60     private static final Map<String, Attribute> attributes;
61  
62     static {
63        final Map<String, Attribute> map = new HashMap<String, Attribute>(64);
64        for (Attribute attribute : values()) {
65           final String name = attribute.getLocalName();
66           if (name != null) {
67              map.put(name, attribute);
68           }
69        }
70        attributes = map;
71     }
72  
73     public static Attribute forName(final String localName) {
74        final Attribute attribute = attributes.get(localName);
75        return attribute == null ? UNKNOWN : attribute;
76     }
77  }
