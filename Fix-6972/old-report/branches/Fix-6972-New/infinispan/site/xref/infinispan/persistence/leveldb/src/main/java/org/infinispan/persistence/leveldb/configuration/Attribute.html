1   package org.infinispan.persistence.leveldb.configuration;
2   
3   import java.util.HashMap;
4   import java.util.Map;
5   
6   /**
7    * Enumerates the attributes used by the LevelDB cache stores configuration
8    * 
9    * @author <a href="mailto:rtsang@redhat.com">Ray Tsang</a>
10   */
11  public enum Attribute {
12     // must be first
13     UNKNOWN(null),
14  
15     BLOCK_SIZE("block-size"),
16     CACHE_SIZE("cache-size"),
17     CLEAR_THRESHOLD("clear-threshold"),
18     COMPRESSION_TYPE("compressionType"),
19     EXPIRED_LOCATION("expiredLocation"),
20     EXPIRY_QUEUE_SIZE("expiryQueueSize"),
21     IMPLEMENTATION_TYPE("implementationType"),
22     LOCATION("location"),
23     PATH("path"),
24     RELATIVE_TO("relative-to"),
25     QUEUE_SIZE("queue-size"),
26     TYPE("type"),
27     ;
28  
29     private final String name;
30  
31     private Attribute(final String name) {
32        this.name = name;
33     }
34  
35     /**
36      * Get the local name of this element.
37      * 
38      * @return the local name
39      */
40     public String getLocalName() {
41        return name;
42     }
43  
44     private static final Map<String, Attribute> attributes;
45  
46     static {
47        final Map<String, Attribute> map = new HashMap<String, Attribute>(64);
48        for (Attribute attribute : values()) {
49           final String name = attribute.getLocalName();
50           if (name != null) {
51              map.put(name, attribute);
52           }
53        }
54        attributes = map;
55     }
56  
57     public static Attribute forName(final String localName) {
58        final Attribute attribute = attributes.get(localName);
59        return attribute == null ? UNKNOWN : attribute;
60     }
61  }
