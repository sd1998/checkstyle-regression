8   public enum AvailabilityMode {
9      /** Regular operation mode */
10     AVAILABLE,
11     /** Data can not be safely accessed because of a network split or successive nodes leaving. */
12     DEGRADED_MODE,
13     /** @deprecated Use {@link #DEGRADED_MODE} instead. */
14     UNAVAILABLE,
15     ;
16  
17     public AvailabilityMode min(AvailabilityMode other) {
18        if (this == UNAVAILABLE || other == UNAVAILABLE)
19           return UNAVAILABLE;
20        if (this == DEGRADED_MODE || other == DEGRADED_MODE)
21           return DEGRADED_MODE;
22        return AVAILABLE;
23     }
24  }
