146     private final String name;
147     private final AttributeDefinition definition;
148 
149     private Attribute(final String name) {
150         this.name = name;
151         this.definition = null;
152     }
153 
154     private Attribute(final AttributeDefinition definition) {
155         this.name = definition.getXmlName();
156         this.definition = definition;
157     }
158 
159     /**
160      * Get the local name of this element.
161      *
162      * @return the local name
163      */
164     public String getLocalName() {
165         return name;
166     }
167 
168     public AttributeDefinition getDefinition() {
169         return definition;
170     }
171 
172     private static final Map<String, Attribute> attributes;
173 
174     static {
175         final Map<String, Attribute> map = new HashMap<String, Attribute>();
176         for (Attribute attribute : values()) {
177             final String name = attribute.getLocalName();
178             if (name != null) map.put(name, attribute);
179         }
180         attributes = map;
181     }
182 
183     public static Attribute forName(String localName) {
184         final Attribute attribute = attributes.get(localName);
185         return attribute == null ? UNKNOWN : attribute;
186     }
