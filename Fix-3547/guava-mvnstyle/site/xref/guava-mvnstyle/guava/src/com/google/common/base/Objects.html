15  package com.google.common.base;
16  
17  import com.google.common.annotations.GwtCompatible;
18  import java.util.Arrays;
19  import org.checkerframework.checker.nullness.compatqual.NullableDecl;
20  
21  /**
22   * Helper functions that can operate on any {@code Object}.
23   *
24   * <p>See the Guava User Guide on <a
25   * href="https://github.com/google/guava/wiki/CommonObjectUtilitiesExplained">writing {@code Object}
26   * methods with {@code Objects}</a>.
27   *
28   * @author Laurence Gonsalves
29   * @since 2.0
30   */
31  @GwtCompatible
32  public final class Objects extends ExtraObjectsMethodsForWeb {
33    private Objects() {}
34  
35    /**
36     * Determines whether two possibly-null objects are equal. Returns:
37     *
38     * <ul>
39     *   <li>{@code true} if {@code a} and {@code b} are both null.
40     *   <li>{@code true} if {@code a} and {@code b} are both non-null and they are equal according to
41     *       {@link Object#equals(Object)}.
42     *   <li>{@code false} in all other situations.
43     * </ul>
44     *
45     * <p>This assumes that any non-null objects passed to this function conform to the {@code
46     * equals()} contract.
47     *
48     * <p><b>Note for Java 7 and later:</b> This method should be treated as deprecated; use {@link
49     * java.util.Objects#equals} instead.
50     */
51    public static boolean equal(@NullableDecl Object a, @NullableDecl Object b) {
52      return a == b || (a != null && a.equals(b));
53    }
54  
55    /**
56     * Generates a hash code for multiple values. The hash code is generated by calling {@link
57     * Arrays#hashCode(Object[])}. Note that array arguments to this method, with the exception of a
58     * single Object array, do not get any special handling; their hash codes are based on identity
59     * and not contents.
60     *
61     * <p>This is useful for implementing {@link Object#hashCode()}. For example, in an object that
62     * has three properties, {@code x}, {@code y}, and {@code z}, one could write:
63     *
64     * <pre>{@code
65     * public int hashCode() {
66     *   return Objects.hashCode(getX(), getY(), getZ());
67     * }
68     * }</pre>
69     *
70     * <p><b>Warning:</b> When a single object is supplied, the returned hash code does not equal the
71     * hash code of that object.
72     *
73     * <p><b>Note for Java 7 and later:</b> This method should be treated as deprecated; use {@link
74     * java.util.Objects#hash} instead.
75     */
76    public static int hashCode(@NullableDecl Object... objects) {
77      return Arrays.hashCode(objects);
78    }
79  }
