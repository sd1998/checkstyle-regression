147     private String prompt() {

149                 + ", designator=" + getCodingSchemeDesignator() + ", version="
150                 + getCodingSchemeVersion() + ", meaning=" + getCodeMeaning()
151                 + "]";
152     }

161         setCodeValue(value);

163         setCodingSchemeVersion(version);
164         setCodeMeaning(meaning);
165         return null;
166     }

170         log.info("Created " + prompt());
171 
172     }

175         log.info("Deleting " + prompt());
176     }

178     public static CodeLocal valueOf(CodeLocalHome codeHome, Dataset item)
179             throws CreateException, FinderException {
180         if (item == null) return null;
181 
182         final String value = item.getString(Tags.CodeValue);
183         final String designator = item.getString(Tags.CodingSchemeDesignator);
184         final String version = item.getString(Tags.CodingSchemeVersion);
185         final String meaning = item.getString(Tags.CodeMeaning);
186         Collection c = codeHome.findByValueAndDesignator(value, designator);
187         for (Iterator it = c.iterator(); it.hasNext();) {
188             final CodeLocal code = (CodeLocal) it.next();
189             if (version == null) { return code; }
190             final String version2 = code.getCodingSchemeVersion();
191             if (version2 == null || version2.equals(version)) { return code; }
192         }
193         return codeHome.create(value, designator, version, meaning);
194     }
195     
196     public static void addCodesTo(CodeLocalHome codeHome, DcmElement sq, Collection c) 
197     throws CreateException, FinderException {
198     	if (sq == null || sq.isEmpty()) return;
199         Dataset item = sq.getItem(0);
200         if (item.isEmpty()) return;
201         c.add(CodeBean.valueOf(codeHome, item));
202     	for (int i = 1, n = sq.countItems(); i < n; i++) {
203     		c.add(CodeBean.valueOf(codeHome, sq.getItem(i)));
204     	}
205     }
206     
207     public static boolean checkCodes(String prompt, DcmElement sq) {
208         if (sq == null || sq.isEmpty())
209             return true;
210         for (int i = 0, n = sq.countItems(); i < n; i++) {
211             Dataset item = sq.getItem(i);
212             if (!item.containsValue(Tags.CodeValue)) {
213                 log.warn("Missing Code Value (0008,0100) in " + prompt
214                         + " - ignore all items");
215                 return false;
216             }
217             if (!item.containsValue(Tags.CodingSchemeDesignator)) {
218                 log.warn("Missing Coding Scheme Designator (0008,0102) in "
219                         + prompt + " - ignore all items");
220                 return false;
221             }
222             if (!item.containsValue(Tags.CodeMeaning)) {
223                 log.warn("Missing Code Meaning (0008,0104) in " + prompt
224                         + " - ignore all items");
225                 return false;
226             }
227         }
228         return true;
229     }
